#3D genomics

1. from .paired.gz to hic

#!/usr/bin/env bash
set -euo pipefail

# ─── User settings ──────────────────────────────────────────────────────────
data_dir="../data"
sample="HAP-WT"

chromsizes="../annotations/hg38.chrom.sizes"
build="hg38"
juicer_tools="../tools/juicer_tools.2.20.00.jar"

merged_pairs="${data_dir}/${sample}_330M.pairs.gz"
sorted_merged="${merged_pairs%.pairs.gz}.sorted.pairs.gz"
dedup_pairs="${data_dir}/${sample}_330M.nodups.pairs.gz"
sorted_dedup="${dedup_pairs%.pairs.gz}.sorted.pairs.gz"
pre_pairs="${data_dir}/${sample}_330M.forPre.pairs.gz"
hic_file="${data_dir}/${sample}_330M.hic"

# comma‐sep list of resolutions (no spaces!)
res_list="2500000,1000000,500000,250000,100000,50000,25000,10000,5000,1000"

threads=12
# ────────────────────────────────────────────────────────────────────────────

echo "2 sort merged pairs"
pairtools sort --nproc $threads -o "$sorted_merged" "$merged_pairs"

echo "3 global deduplication"
pairtools dedup --nproc-in  $threads --nproc-out $threads --output "$dedup_pairs" "$sorted_merged"

echo "4 sort deduplicated pairs"
pairtools sort --nproc $threads -o "$sorted_dedup" "$dedup_pairs"

echo "5 prepare for juicebox pre:"
gunzip -c "$sorted_dedup" | grep -E '^(## pairs format|^#columns|^[^#])' | gzip -c > "$pre_pairs"

echo "6 building multi-res .hic"
java -Xmx16g -jar "$juicer_tools" pre -t $threads -r "$res_list" "$pre_pairs" "$hic_file" "$build"
